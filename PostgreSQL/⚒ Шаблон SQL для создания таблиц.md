# ⚒ Шаблон SQL для создания таблиц

## Назначение

Данный SQL шаблон предназначен для ускорения разработки. Скопировал, удалил лишнее, добавил нужное. Готово. Он соответствует следующим требованиям:
1. Правила по именованию объектов БД
1. Правила именований объектов на английском языке
1. Правила создания таблиц БД
1. Создание изменений для БД

## Определение таблицы

`{schema}`, `{table}`, `{ref_table}`, `{column}`, `{role_old}`, `{role_new}` — это шаблоны названий (схемы, таблицы, связанной таблицы, колонки, роли новой, роли старой), вместо которых нужно подставить настоящие названия.

> [!CAUTION]
> Формат хранения данных в БД никак не должен зависеть от вида отображаемых данных и способа их ввода в клиентском ПО (браузер, моб. приложение и др.)! Например, номер телефона в макете дизайна (не) может содержать пробелы или знаки подчёркивания. Но в БД его формат номера телефона должен быть строго единым. Дизайн может поменяться, а данные в БД будут продолжать храниться годами. Для гарантии единого формата используйте валидацию на уровне БД: выражение CHECK() или домены.

> [!TIP]
> Какое значение по умолчанию лучше использовать для необязательных полей — NULL или пустую строку / 0 / пустой массив?
> Для однозначности и минимизации ошибок лучше выбрать только один способ хранения.
> В PostgreSQL хранить `NULL` выгоднее любых других типов значений
> В PHP в параметрах методов или возвращаемых значениях методов nullable тип `?string / ?int / ?array` выглядит понятнее, чем без знака вопроса.

> [!WARNING]
> Почему в изменении БД опасно использовать синтаксис CREATE TABLE IF NOT EXISTS {table}?
> Если в процессе разработки вы измените схему данных таблицы, то SQL запрос на создание таблицы будет проигнорирован, если такая таблица уже существует. Дальнейшее тестирование таблицы с устаревшей структурой будет некорректным. А если такое тестирование будет успешным, есть риск отправить в производственную среду нерабочее решение. Поэтому лучше всего делать так:
> 1. Удалить таблицу, если она существует
> 1. Создать таблицу
> 1. Наполнить таблицу первичными данными (при необходимости)
> 1. Выполнить тесты (при необходимости)

```sql
DROP TABLE IF EXISTS {table};
 
CREATE TABLE {table} -- IF EXISTS не используем, почему - см. выше
(
    -- If you use GENERATED ALWAYS AS IDENTITY, you will get an error message if you try to override the generated value by explicitly inserting a number. This avoids the common problem that manually entered values will conflict with generated values later on, causing surprising application errors.
	id 					integer			GENERATED ALWAYS AS IDENTITY PRIMARY KEY, -- Самый правильный способ, см. https://web.archive.org/web/20230206190645/https://www.depesz.com/2017/04/10/waiting-for-postgresql-10-identity-columns/
 	id 					integer			GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, -- Если у вас ORM Doctrine старых версий и вы получаете ошибку ERROR: cannot insert a non-DEFAULT value into column "id". DETAIL:  Column "id" is an identity column defined as GENERATED ALWAYS.
	id 					serial  		CONSTRAINT {table}_pk PRIMARY KEY, -- УСТАРЕЛО, НЕ ИСПОЛЬЗОВАТЬ! Нужно использовать IDENTITY, см. выше
	guid				uuid			PRIMARY KEY, -- https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/uuid_generate_v7.sql
	-- текстовые поля на доменах (валидация скрыта внутри доменов)
    -- не добавляйте колонки с доменами в большие таблицы, т.к. в этом случае ALTER TABLE {table} ADD COLUMN {column}... работает медленно
	-- email и phone имеют минимальную валидацию, чтобы работало быстро, для полной валидации см. https://github.com/rin-nas/postgresql-patterns-library/blob/master/README.md#как-провалидировать-значение-поля-только-если-оно-явно-указано-в-update-запросе
	email        		email   		DEFAULT NULL,	-- https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/email.sql
	phone				phone			DEFAULT NULL, 	-- https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/phone.sql
	inn					inn10			DEFAULT NULL, 	-- ИНН юридического лица https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/inn10.sql 
	inn					inn12			DEFAULT NULL, 	-- ИНН физического лица и ИП https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/inn12.sql 
	inn					inn				DEFAULT NULL, 	-- ИНН юр. или физ. лица или ИП https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/inn.sql
	kpp					kpp			 	DEFAULT NULL,	-- КПП https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/kpp.sql
    ogrn				ogrn			DEFAULT NULL,	-- ОГРН https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/ogrn.sql
    ogrnip				ogrnip			DEFAULT NULL,	-- ОГРНИП https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/ogrnip.sql
    bik					bik			 	DEFAULT NULL,	-- БИК https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/bik.sql
    client_acc			client_account	DEFAULT NULL,	-- Банковский расчётный (клиентский) счёт (р/с) https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/client_account.sql
    corresp_acc			correspondent_account DEFAULT NULL,	-- Банковский корреспонденткий счёт (к/с) https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/correspondent_account.sql
	kladr				kladr			DEFAULT NULL,	-- КЛАДР --https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/kladr.sql
	color				css_color 		NOT NULL, 		-- CSS цвет https://github.com/rin-nas/postgresql-patterns-library/blob/master/domains/css_color.sql
   
	-- текстовые поля не на доменах (необходимо явное указание валидации в CHECK)
	name        		text 			NOT NULL		CHECK(LENGTH(name) BETWEEN 1 AND 255 AND trim(name) = name), -- проверка обязательности заполнения и отсутствия лишних начальных и конечных пробелов
	name_translit		text			NOT NULL        CHECK(LENGTH(name_translit) BETWEEN 1 AND 255 AND trim(name_translit) = name_translit), -- название с транслитерацией, используйте SQL функцию iuliia_translate_mosmetro()
	description			text			DEFAULT NULL 	CHECK(LENGTH(description) BETWEEN 1 AND 10000 AND trim(description) = description),
	phone_extension		text			DEFAULT NULL	CHECK(LENGTH(phone_extension) BETWEEN 1 AND 20 AND phone_extension ~ '^\d+$'), --локальный номер телефона в офисе
	image_url			text   			DEFAULT NULL	CHECK(LENGTH(image_url) BETWEEN 1 AND 2048 AND image_url ~ '^[\x21-\x7e]+$'), -- URL к картинке с логотипом, например '/path/to/filename.svg' ('^[\x21-\x7e]+$' is part of ASCII without space and control characters)
	external_url		text		 	DEFAULT NULL 	CHECK(LENGTH(external_url) BETWEEN 1 AND 2048 AND external_url ~ '^https?://[\x21-\x7e\w]+$'), -- URL с обязательным протоколом и с поддержкой русских букв
    image_path			text		 	DEFAULT NULL	CHECK(LENGTH(image_path) BETWEEN 1 AND 255 AND image_path ~ '^\d{4}\/\d{2}\/\d{2}\/(?!-)(?!.+?--)[-\da-z]+(?<!-)\.[\da-z]{3,4}$'), -- путь к картинке с датой, например, '2018/07/03/background-tele2-kaluga-105.jpg'
    image_path			text			DEFAULT NULL	CHECK(LENGTH(image_path) BETWEEN 1 AND 255 AND image_path ~ '^\/\d{4}\/\d{2}\/\d{2}\/[\da-f]{13}\.[\da-z]{3,4}$'), -- путь к картинке с датой, например, '/2021/05/11/609a42f5debb3.png' где 609a42f5debb3 - это результат PHP функции uniqid()
	slug                text		    NOT NULL 		CHECK(LENGTH(slug) BETWEEN 1 AND 255 AND slug ~ '^[a-z\d\-_]+$'), -- see SQL function iuliia_translate_mosmetro() and old slugify()  
	description 		text  			DEFAULT NULL 	CHECK(LENGTH(description) BETWEEN 1 AND 10000 AND trim(description) = description),
	password_hash		text			DEFAULT NULL 	CHECK(OCTET_LENGTH(password_hash) = 60), -- https://wiki.rabota.space/pages/viewpage.action?pageId=47284308
	sql_query           text            DEFAULT NULL    check(is_sql(sql_query) AND trim(sql_query) = sql_query), --поле для хранения SQL запроса к PostgreSQL
	orgn				text 			DEFAULT NULL	check(is_ogrn(orgn)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_ogrn.sql
	orgnip				text 			DEFAULT NULL	check(is_ogrn(orgnip)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_ogrnip.sql
	bik					text			DEFAULT NULL	check(is_bik(bik)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_bik.sql
	kpp					text			DEFAULT NULL	check(is_kpp(kpp)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_kpp.sql
	client_acc			text		 	DEFAULT NULL	check(is_client_account(client_account)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_client_account.sql
	corresp_acc			text		 	DEFAULT NULL	check(is_correspondent_account(corresp_account)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_correspondent_account.sql
	snils				text			DEFAULT NULL	check(is_snils(snils)), --https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/is_snils.sql
 
	sort_number	        integer         DEFAULT NULL,	-- сортировка записей таблицы (select * from unnest(array[2,7, null, 5, null]) as t(x) order by x asc nulls first) 
	salary				integer			DEFAULT NULL	CHECK(salary > 0), -- зарплата 
    price               money           DEFAULT NULL, 	-- цена, https://postgrespro.ru/docs/postgresql/16/datatype-money
    price               numeric(15,2)   DEFAULT NULL, 	-- цена, если тип money со своим форматированием не подходит
    cross_rate          numeric(15,5)   DEFAULT NULL, 	-- кросс-курс (отношение одной денежной единицы к другой)
	--https://stackoverflow.com/questions/13837258/what-is-an-appropriate-data-type-to-store-a-timezone
 	time_zone           interval        DEFAULT NULL	CHECK(time_zone between '-12:00'::interval and '14:00'::interval), --https://en.wikipedia.org/wiki/Time_zone
    is_hidden           boolean         DEFAULT FALSE	NOT NULL,
    is_visible          boolean         DEFAULT TRUE	NOT NULL,
	birthday 		    date 		    DEFAULT NULL	CHECK(EXTRACT(YEAR FROM age(birthday)) between 7 and 100), -- ограничение на возраст
 
    -- Для полей с WITH TIME ZONE PostgreSQL внутри таблиц хранит дату-время в UTC+0:00. 
    -- А часовые пояса БД как раз использует для корректной работы при записи (конвертация из указанного часового пояса в UTC+0:00) 
    -- и чтении (из UTC+0:00 в, например, UTC+3:00 - если я нахожусь в Москве).
    -- Для without time zone нужно добавить комментарий с обоснованием, почему так!
    created_at 		    timestamp(0) 	WITH TIME ZONE	DEFAULT NOW() NOT NULL CHECK(created_at <= now() + interval '10m'), -- учитываем, что время на серверах для php и БД может незначительно отличаться
    -- поле updated_at может обновляться автоматически триггером, см. готовый код триггера ниже
    updated_at 			timestamp(0) 	WITH TIME ZONE	DEFAULT NOW() NOT NULL CHECK(updated_at <= now() + interval '10m'), -- учитываем, что время на серверах для php и БД может незначительно отличаться
 
    deleted_at			timestamp(0) 	WITH TIME ZONE  DEFAULT NULL CHECK(deleted_at <= now() + interval '10m'), -- запись считается [не]удалённой для условия deleted_at is [not] null,
 
    coordinates         point			DEFAULT NULL CHECK(coordinates[0]/*longitude=y*/ between -180 and 180 AND coordinates[1]/*latitude=x*/ between -90 and 90),
    latitude            numeric(9,6)    DEFAULT NULL CHECK(latitude between -90 and 90), -- устарело, см. поле "coordinates", см. INF-1040
    longitute           numeric(9,6)	DEFAULT NULL CHECK(longitute between -180 and 180), -- устарело, см. поле "coordinates", см. INF-1040
 
    passed_steps		bit(128), -- битовый тип годится только для хранения > 64 вариантов значений (битов), иначе используйте smallint/int/bigint!
                                  -- обоснование: select pg_column_size(0::smallint), pg_column_size(0::int), pg_column_size(0::bigint), pg_column_size(0::bit(8)), pg_column_size(0::bit(16))
 
	-- Для таблицы-журнала, см. https://github.com/rin-nas/postgresql-patterns-library#Как-сделать-журналирование-изменений-таблицы
	reference_id integer 		not null, -- идентификатор связанной сущности, FK быть не должно, т.к. запись м.б. удалена
    old_data     jsonb			check(jsonb_typeof(old_data) = 'object'),
    new_data     jsonb			check(jsonb_typeof(new_data) = 'object'),
 
    -- Для нашего API, куда клиенты посылают запросы, а мы отвечаем
    http_request_headers 	jsonb 	check(jsonb_typeof(http_request_headers) = 'array'), -- HTTP заголовки в формате ключ-значение в виде массива объектов (из-за возможных дубликатов заголовков)
	request_remote_addr 	inet, 	-- IP адрес
    session_id				uuid,   -- ID сессии, см. https://github.com/rin-nas/postgresql-patterns-library/blob/master/functions/uuid_generate_v7.sql
	
    -- Для стороннего API, куда мы посылаем запросы и нам отвечают
    http_response_headers 	jsonb,  	check(jsonb_typeof(http_response_headers) = 'array'), -- HTTP заголовки в формате ключ-значение в виде массива объектов (из-за возможных дубликатов заголовков)
    http_response_code  	smallint 	check(response_code between 100 and 599), --код ответа сервера
	http_response_body_hash	uuid,		-- md5, приведённый к типу uuid занимает 16 байт вместо 32 байт
 
	-- Идентификаторы:
    {ref_table}_id			integer		REFERENCES {ref_table} (id) on delete set null, -- идентификатор связанной сущности, например person_id
    {ref_table}_deleted_id 	integer, -- идентификатор связанной удалённой сущности, FK быть не должно, например person_deleted_id
	created_admin_id	integer         REFERENCES v3_admin (id) on delete set null, -- для админки, любой внешний ключ должен иметь индекс!
    updated_admin_id	integer         REFERENCES v3_admin (id) on delete set null, -- для админки, любой внешний ключ должен иметь индекс!
	created_person_id	integer         REFERENCES v3_person (id) on delete set null, -- для клиентов, любой внешний ключ должен иметь индекс!
    updated_person_id	integer         REFERENCES v3_person (id) on delete set null, -- для клиентов, любой внешний ключ должен иметь индекс!
 
    --массив уникальных значений, массив с id это обычно ошибка проектирования, см. https://wiki.rabota.space/pages/viewpage.action?pageId=43974746#id-ПравиласозданиятаблицБД-Массивы
    salaries			integer[]		DEFAULT NULL CHECK(cardinality(salaries) > 0 and 
                                                           cardinality(array_unique(salaries)) = cardinality(salaries)), 
    tree_path_ids       ltree           NOT NULL CHECK(
                                            tree_path_ids::text ~ '^\d+(\.\d+)*$|^$' -- список родительских id через точку
                                            and check_foreign_key_array(string_to_array(tree_path_ids::text, '.')::integer[], '{schema}', '{table}', '{column}')
                                        ),
    -- ограничения уровня таблицы:
	CONSTRAINT {table}_created_at_updated_at CHECK(created_at <= updated_at + interval '10m'), -- учитываем, что время на серверах для php и БД может незначительно отличаться
	CONSTRAINT {table}_deleted_id_check CHECK ( ({ref_table}_id is null and {ref_table}_deleted_id is not null)
											 OR ({ref_table}_id is not null and {ref_table}_deleted_id is null) ),
 
    constraint {table}_client_acc_bik_check check(is_client_account(client_acc, bik)), --валидация р/с по БИК
    constraint {table}_corresp_acc_bik_check check(is_correspondent_account(corresp_acc, bik)), --валидация к/с по БИК
 
    constraint {table}_old_new_data_check check(not(old_data is null and new_data is null)) -- только для таблицы-журнала, см. https://github.com/rin-nas/postgresql-patterns-library#Как-сделать-журналирование-изменений-таблицы
 
	-- Закомментировать, если данные нужно хранить на дорогом и быстром SSD. Предполагаются SQL запросы реального времени, где нужно быстродействие.
	-- Раскомментировать, если данные можно хранить на дешёвом и медленном HDD. Предполагаются SQL запросы не реального времени (по расписанию) или данных очень-очень много или доступ к ним происходит редко.
	-- tablespace slowspace -- ВНИМАНИЕ! пока закомментировать, см. INF-3482
;
 
--для избегания проблем с доступами к объектам БД под разными учётками (например, под {role_new} к реплике)
ALTER SCHEMA {schema} OWNER TO {role_new}; 
ALTER TABLE {table} OWNER TO {role_new};
ALTER VIEW {view} OWNER TO {role_new};
ALTER TYPE {type} OWNER TO {role_new};
ALTER DOMAIN {domain} OWNER TO {role_new};
ALTER FUNCTION {function}(...) OWNER TO {role_new};
ALTER PROCEDURE {procedure}(...) OWNER TO {role_new};
 
-- вместо нескольких команд ALTER ... OWNER TO ... можно использовать одну команду REASSIGN
REASSIGN OWNED BY {role_old} TO {role_new};
 
-- Описания таблиц и колонок не должны быть связаны какими-либо интерфейсами (GUI), это технический слой данных!
COMMENT ON TABLE {table} IS 'Описание таблицы на русском языке';
COMMENT ON COLUMN {table}.{column} IS 'Описание колонки на русском языке';
 
-- при удалении связанной сущности (например, компании или персоны) в соответствующее поле {ref_table}_id записывается null, а айдишник записывается в {ref_table}_deleted_id
create or replace function {table}_before_update() returns trigger
language plpgsql
as
$$
begin
	--обработка срабатывания триггера по ограничению {ref_table}_id REFERENCES {ref_table} (id) ON DELETE SET NULL
    if (new.{ref_table}_id is null and old.{ref_table}_id is not null) 
    then
        new.{ref_table}_deleted_id = old.{ref_table}_id;
    end if;
    return new;
end
$$;
 
drop trigger if exists {table}_before_update on {table};
create trigger {table}_before_update     before update on {table}
    for each row execute procedure {table}_before_update();
 
-- Наполняем таблицу первичными данными, при необходимости
INSERT INTO {table} ({column1}, {column2}, ...) VALUES 
(...), 
(...);
```
